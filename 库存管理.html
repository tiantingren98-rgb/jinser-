(function() {

    'use strict';

    const STORAGE_KEY = 'inventory_data';

    const defaultData = [

        { id: 1, sku: 'TS-001', color: '黑色', size: 'M', stock: 23 },

        { id: 2, sku: 'TS-001', color: '黑色', size: 'L', stock: 15 },

        { id: 3, sku: 'TS-002', color: '白色', size: 'S', stock: 8 },

        { id: 4, sku: 'TS-002', color: '白色', size: 'M', stock: 31 },

        { id: 5, sku: 'SHOE-001', color: '红色', size: '42', stock: 5 },

        { id: 6, sku: 'SHOE-001', color: '红色', size: '43', stock: 12 },

        { id: 7, sku: 'CAP-001', color: '藏青', size: '均码', stock: 18 },

    ];

    let inventory = [];

    let nextId = 1;

    let searchTerm = '', filterColor = '', filterSize = '';

    const tbody = document.getElementById('inventoryBody');

    const totalProductsEl = document.getElementById('totalProducts');

    const totalVariantsEl = document.getElementById('totalVariants');

    const totalStockEl = document.getElementById('totalStock');

    const lowStockCountEl = document.getElementById('lowStockCount');

    const inputSku = document.getElementById('inputSku');

    const inputColor = document.getElementById('inputColor');

    const inputSize = document.getElementById('inputSize');

    const inputStock = document.getElementById('inputStock');

    const addBtn = document.getElementById('addBtn');

    const searchInput = document.getElementById('searchInput');

    const filterColorEl = document.getElementById('filterColor');

    const filterSizeEl = document.getElementById('filterSize');

    const clearFiltersBtn = document.getElementById('clearFilters');

    const exportBtn = document.getElementById('exportBtn');

    const importBtn = document.getElementById('importBtn');

    const fileInput = document.getElementById('fileInput');

    const toastEl = document.getElementById('toast');


    function loadData() {

        const stored = localStorage.getItem(STORAGE_KEY);

        if (stored) {

            try {

                const parsed = JSON.parse(stored);

                if (Array.isArray(parsed) && parsed.length) {

                    inventory = parsed;

                    const maxId = inventory.reduce((max, item) => Math.max(max, item.id || 0), 0);

                    nextId = maxId + 1;

                    return;

                }

            } catch(e) {}

        }

        inventory = defaultData.map(item => ({...item}));

        nextId = inventory.reduce((max, item) => Math.max(max, item.id || 0), 0) + 1;

        saveData();

    }

    function saveData() { localStorage.setItem(STORAGE_KEY, JSON.stringify(inventory)); }

    let toastTimer = null;

    function showToast(message, type='success') {

        toastEl.textContent = message;

        toastEl.className = 'toast ' + type;

        void toastEl.offsetWidth;

        toastEl.classList.add('show');

        clearTimeout(toastTimer);

        toastTimer = setTimeout(() => { toastEl.classList.remove('show'); }, 2200);

    }

    function addItem(sku, color, size, stock) {

        if (!sku.trim() || !color.trim() || !size.trim()) { showToast('请完整填写 SKU、颜色、尺码', 'error'); return false; }

        const stockNum = parseInt(stock) || 0;

        if (stockNum < 0) { showToast('库存不能为负数', 'error'); return false; }

        const skuTrim = sku.trim(), colorTrim = color.trim(), sizeTrim = size.trim();

        const exist = inventory.find(item => item.sku === skuTrim && item.color === colorTrim && item.size === sizeTrim);

        if (exist) { showToast(`该变体已存在 (${skuTrim} · ${colorTrim} · ${sizeTrim})`, 'error'); return false; }

        const skuExists = inventory.some(item => item.sku === skuTrim);

        const newItem = { id: nextId++, sku: skuTrim, color: colorTrim, size: sizeTrim, stock: stockNum };

        inventory.push(newItem);

        saveData();

        renderAll();

        showToast(skuExists ? `✅ 已为 SKU「${skuTrim}」添加新变体 (${colorTrim} · ${sizeTrim})` : `✅ 已添加新产品 SKU「${skuTrim}」`);

        return true;

    }

    function deleteItem(id) {

        const item = inventory.find(i => i.id === id);

        if (!item) return;

        if (!confirm(`确定要删除变体「${item.sku} · ${item.color} · ${item.size}」吗？`)) return;

        inventory = inventory.filter(i => i.id !== id);

        saveData();

        renderAll();

        showToast('已删除该变体', 'success');

    }

    function updateStock(id, delta) {

        const item = inventory.find(i => i.id === id);

        if (!item) return;

        const newStock = item.stock + delta;

        if (newStock < 0) { showToast('库存不能为负数', 'error'); return; }

        item.stock = newStock;

        saveData();

        renderAll();

    }

    function exportData() {

        if (inventory.length === 0) { showToast('暂无数据可导出', 'error'); return; }

        const dataStr = JSON.stringify(inventory, null, 2);

        const blob = new Blob([dataStr], { type: 'application/json' });

        const url = URL.createObjectURL(blob);

        const a = document.createElement('a');

        a.href = url;

        a.download = `inventory_backup_${new Date().toISOString().slice(0,10)}.json`;

        document.body.appendChild(a);

        a.click();

        document.body.removeChild(a);

        URL.revokeObjectURL(url);

        showToast('✅ 导出成功，请保存好文件');

    }

    function importData(file) {

        const reader = new FileReader();

        reader.onload = function(e) {

            try {

                const imported = JSON.parse(e.target.result);

                if (!Array.isArray(imported) || imported.length === 0) { showToast('文件格式错误或为空', 'error'); return; }

                if (!imported[0].sku || !imported[0].color || !imported[0].size) { showToast('数据格式不匹配，请检查文件', 'error'); return; }

                if (!confirm(`即将导入 ${imported.length} 条数据，当前数据将被覆盖。是否继续？`)) return;

                inventory = imported.map((item, index) => ({...item, id: nextId + index}));

                nextId = nextId + imported.length;

                saveData();

                renderAll();

                showToast(`✅ 成功导入 ${imported.length} 条数据`);

            } catch(err) { showToast('导入失败：文件解析错误', 'error'); console.error(err); }

        };

        reader.readAsText(file);

        fileInput.value = '';

    }

    function getFilteredData() {

        let result = [...inventory];

        if (searchTerm.trim()) { const term = searchTerm.trim().toLowerCase(); result = result.filter(item => item.sku.toLowerCase().includes(term) || item.color.toLowerCase().includes(term) || item.size.toLowerCase().includes(term)); }

        if (filterColor) { result = result.filter(item => item.color === filterColor); }

        if (filterSize) { result = result.filter(item => item.size === filterSize); }

        return result;

    }

    function renderStats(data) {

        const uniqueSkus = new Set(data.map(d => d.sku));

        totalProductsEl.textContent = uniqueSkus.size;

        totalVariantsEl.textContent = data.length;

        totalStockEl.textContent = data.reduce((sum, item) => sum + item.stock, 0);

        lowStockCountEl.textContent = data.filter(item => item.stock <= 5).length;

    }

    function getColorHex(color) {

        const map = { '黑色':'#1a1a2e','白色':'#f0f0f0','红色':'#e53e3e','蓝色':'#2a6df4','绿色':'#0f973d','黄色':'#d69e2e','紫色':'#805ad5','橙色':'#ed8936','粉色':'#ed64a6','灰色':'#a0aec0','藏青':'#1e2a4a','棕色':'#8b6b4b','米色':'#f5e6d3','金色':'#d4af37','银色':'#c0c0c0' };

        return map[color] || '#cccccc';

    }

    function escHtml(str) { const div = document.createElement('div'); div.textContent = str; return div.innerHTML; }

    function renderTable(data) {

        if (data.length === 0) {

            tbody.innerHTML = `<tr><td colspan="6"><div class="empty-state"><div class="big-icon">📭</div><h3>暂无匹配的库存记录</h3><p style="color:#7f8ea8;">尝试调整筛选条件，或点击上方「新增变体」添加</p></div></td></tr>`;

            return;

        }

        const groups = {};

        data.forEach(item => { if (!groups[item.sku]) groups[item.sku] = []; groups[item.sku].push(item); });

        const sortedSkus = Object.keys(groups).sort();

        let html = '';

        sortedSkus.forEach(sku => {

            const items = groups[sku];

            items.sort((a,b) => a.color.localeCompare(b.color) || a.size.localeCompare(b.size));

            items.forEach((item, index) => {

                const stockClass = item.stock <= 5 ? 'stock-low' : (item.stock <= 15 ? 'stock-medium' : 'stock-high');

                const colorStyle = `background-color: ${getColorHex(item.color)};`;

                html += `<tr>`;

                if (index === 0) html += `<td rowspan="${items.length}" class="sku-group-cell" style="vertical-align: middle;">${escHtml(sku)}</td>`;

                html += `<td><span class="color-dot" style="${colorStyle}"></span>${escHtml(item.color)}</td><td><span class="size-tag">${escHtml(item.size)}</span></td><td style="text-align:center;"><span class="stock-num ${stockClass}">${item.stock}</span></td><td style="text-align:center;"><div class="action-group"><button class="btn btn-success btn-sm" data-action="inc" data-id="${item.id}">+1</button><button class="btn btn-danger btn-sm" data-action="dec" data-id="${item.id}">−1</button><button class="btn btn-outline btn-sm" data-action="set" data-id="${item.id}">✎</button></div></td><td style="text-align:right;"><button class="btn-icon danger" data-action="delete" data-id="${item.id}" title="删除变体">🗑</button></td></tr>`;

            });

        });

        tbody.innerHTML = html;

        tbody.querySelectorAll('[data-action]').forEach(btn => {

            btn.addEventListener('click', function(e) {

                e.stopPropagation();

                const action = this.dataset.action;

                const id = parseInt(this.dataset.id);

                if (action === 'delete') deleteItem(id);

                else if (action === 'inc') updateStock(id, 1);

                else if (action === 'dec') updateStock(id, -1);

                else if (action === 'set') {

                    const item = inventory.find(i => i.id === id);

                    if (!item) return;

                    const newVal = prompt(`设置 ${item.sku} · ${item.color} · ${item.size} 的库存数量:`, item.stock);

                    if (newVal === null) return;

                    const num = parseInt(newVal);

                    if (isNaN(num) || num < 0) { showToast('请输入有效的非负整数', 'error'); return; }

                    item.stock = num;

                    saveData();

                    renderAll();

                    showToast(`✅ 已更新库存为 ${num}`);

                }

            });

        });

    }

    function updateFilterOptions() {

        const colors = [...new Set(inventory.map(item => item.color))].sort();

        const currentColor = filterColor;

        filterColorEl.innerHTML = '<option value="">所有颜色</option>' + colors.map(c => `<option value="${escHtml(c)}" ${c === currentColor ? 'selected' : ''}>${escHtml(c)}</option>`).join('');

        const sizes = [...new Set(inventory.map(item => item.size))].sort();

        const currentSize = filterSize;

        filterSizeEl.innerHTML = '<option value="">所有尺码</option>' + sizes.map(s => `<option value="${escHtml(s)}" ${s === currentSize ? 'selected' : ''}>${escHtml(s)}</option>`).join('');

    }

    function renderAll() {

        const data = getFilteredData();

        renderStats(data);

        renderTable(data);

        updateFilterOptions();

        filterColorEl.value = filterColor;

        filterSizeEl.value = filterSize;

    }


    addBtn.addEventListener('click', function() {

        const sku = inputSku.value, color = inputColor.value, size = inputSize.value, stock = inputStock.value;

        const success = addItem(sku, color, size, stock);

        if (success) { inputSku.value = ''; inputColor.value = ''; inputSize.value = ''; inputStock.value = '10'; inputSku.focus(); }

    });

    document.querySelectorAll('#inputSku, #inputColor, #inputSize, #inputStock').forEach(el => {

        el.addEventListener('keydown', function(e) { if (e.key === 'Enter') addBtn.click(); });

    });

    searchInput.addEventListener('input', function() { searchTerm = this.value; renderAll(); });

    filterColorEl.addEventListener('change', function() { filterColor = this.value; renderAll(); });

    filterSizeEl.addEventListener('change', function() { filterSize = this.value; renderAll(); });

    clearFiltersBtn.addEventListener('click', function() { searchInput.value = ''; searchTerm = ''; filterColor = ''; filterSize = ''; renderAll(); showToast('已清除所有筛选', 'success'); });

    exportBtn.addEventListener('click', exportData);

    importBtn.addEventListener('click', function() { fileInput.click(); });

    fileInput.addEventListener('change', function(e) { if (this.files && this.files.length > 0) importData(this.files[0]); });

    document.querySelector('.stats')?.addEventListener('dblclick', function() { clearFiltersBtn.click(); });


    loadData();

    renderAll();

    console.log('📦 库存管理 (共享版) 已启动 — 支持导入导出');

})();
